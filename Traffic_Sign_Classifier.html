from skimage import exposure

def pre_process_data(X):
    # Converting to grayscale
    X = 0.299 * X[:, :, :, 0] + 0.587 * X[:, :, :, 1] + 0.114 * X[:, :, :, 2]
    # Normalizing dataset
    X = (X / 255.).astype(np.float32)
    # Applying histogram equalization for better contrast.
    for i in range(X.shape[0]):
        X[i] = exposure.equalize_adapthist(X[i])
    X = X.reshape(X.shape + (1,)) 
    return X

import tensorflow as tf

def Convnet(x, keep_prob, mu=0, sigma=0.01):    
    #Conv1
    with tf.variable_scope('conv1'):
    
        wc1 = tf.Variable(tf.truncated_normal((5, 5, 1, 6), mean = mu, stddev = sigma), name = 'wc1')
        bc1 = tf.Variable(tf.zeros(6), name = 'bc1')
        conv1 = tf.nn.conv2d(x, filter = wc1, strides = [1, 1, 1, 1], padding = 'VALID')
        conv1 = tf.nn.bias_add(conv1, bc1)
    
        #Batch_normalization
        conv1_norm = tf.contrib.layers.batch_norm(conv1)
    
    #Activation
    with tf.variable_scope('conv1_activation'):
        conv1_activation = tf.nn.relu(conv1_norm)
    
    #Max Pool
    with tf.variable_scope('conv1_pool'):
        conv1_pool = tf.nn.max_pool(conv1_activation, ksize = [1, 2, 2, 1], strides = [1, 2, 2, 1], padding = 'VALID')
    
    #Conv2
    with tf.variable_scope('conv2'):
        wc2 = tf.Variable(tf.truncated_normal((5, 5, 6, 16), mean = mu, stddev = sigma), name = 'wc2')
        bc2 = tf.Variable(tf.zeros(16), name = 'bc2')
        conv2 = tf.nn.conv2d(conv1_pool, filter = wc2, strides = [1, 1, 1, 1], padding = 'VALID')
        conv2 = tf.nn.bias_add(conv2, bc2)
    
        #Batch_normalization
        conv2_norm = tf.contrib.layers.batch_norm(conv2)
    
    #Activation
    with tf.variable_scope('conv2_activation'):
        conv2_activation = tf.nn.relu(conv2_norm)
    
    #Max Pool
    with tf.variable_scope('conv2_pool'):
        conv2_pool = tf.nn.max_pool(conv2_activation, ksize = [1, 2, 2, 1], strides = [1, 2, 2, 1], padding = 'VALID')
    
    #Flatten the output
    fc0 = tf.contrib.layers.flatten(conv2_pool)
    
    #First Fully Connected Layer
    with tf.variable_scope('fc1'):
        wd1 = tf.Variable(tf.truncated_normal((400, 120), mean = mu, stddev = sigma), name = 'wd1')
        bd1 = tf.Variable(tf.zeros(120), name = 'bd1')
        fc1 = tf.add(tf.matmul(fc0, wd1), bd1)
    
        #Batch_normalization
        fc1 = tf.contrib.layers.batch_norm(fc1)
        
        #Activation
        fc1 = tf.nn.relu(fc1)
    
        #Dropout
        fc1 = tf.nn.dropout(fc1, keep_prob)
    
    #Second Fully Connected Layer
    with tf.variable_scope('fc2'):
        wd2 = tf.Variable(tf.truncated_normal((120, 84), mean = mu, stddev = sigma), name = 'wd2')
        bd2 = tf.Variable(tf.zeros(84), name = 'bd2')
        fc2 = tf.add(tf.matmul(fc1, wd2), bd2)
    
        #Batch_normalization
        fc2 = tf.contrib.layers.batch_norm(fc2)
    
        #Activation
        fc2 = tf.nn.relu(fc2)
    
        #Dropout
        fc2 = tf.nn.dropout(fc2, keep_prob)
    
    #Third Fully Connected Layer
    with tf.variable_scope('out'):
        wd3 = tf.Variable(tf.truncated_normal((84, n_classes), mean = mu, stddev = sigma), name = 'wd3')
        bd3 = tf.Variable(tf.zeros(n_classes), name = 'bd3')
        logits = tf.add(tf.matmul(fc2, wd3), bd3)
    
    return logits, [conv1, conv2]

Lets do a sanity check that we implemented the architecture correclty.¶

Initially when we start our model should be unbiased and should return equal probability for any prediction. Initially all the logits will be nearly equal so softmax probability for each class will be (1/n_classes). When we calculate the total_loss initially it will be -ln(1/n_classes). In our case n_classes is 43 so the value of -ln(1/n_classes) will be ~3.76. Let us validate our initial cost to check if this the case.

Also we will use summary writer to create a image of computational graph. We will use tensorboard to get the computational graph image.

# Lets write a function to generate batches of data.
# We will use python generators with yield
def get_batches(X_train, y_train, batch_size):
    n_examples = np.shape(X_train)[0]
    for offset in range(0, n_examples, batch_size):
        start_idx = offset
        end_idx = offset + batch_size
        yield X_train[start_idx:end_idx], y_train[start_idx:end_idx]

def evaluate_accuracy(sess, X_data, y_data, predictions, one_hot_y):
    correct_prediction = tf.equal(tf.argmax(predictions, 1), tf.argmax(one_hot_y, 1))
    accuracy_op = tf.reduce_mean(tf.cast(correct_prediction, tf.float32))
    feed_dict = {x: X_data, y: y_data, keep_prob: 1.0}
    accuracy = sess.run(accuracy_op, feed_dict = feed_dict)
    return accuracy

from sklearn.utils import shuffle

init = tf.global_variables_initializer()
saver = tf.train.Saver()
with tf.Session() as sess:
    sess.run(init)
    writer.add_graph(sess.graph)
    initial_cost = sess.run(cost, feed_dict = {x: X_train_normalized, y: y_train, keep_prob: 1.0})

    for epoch in range(epochs):
        X_train_normalized, y_train = shuffle(X_train_normalized, y_train)
        batches = get_batches(X_train_normalized, y_train, batch_size)
        for X_batch, y_batch in batches:
            feed_dict = {x: X_batch, y: y_batch, keep_prob: 0.5}
            sess.run(train_op, feed_dict = feed_dict)
        current_cost, summary_result = sess.run([cost, summary], feed_dict = {x: X_train_normalized, y: y_train, keep_prob: 1.0})
        writer.add_summary(summary_result, epoch)
        train_accuracy = evaluate_accuracy(sess, X_train_normalized, y_train, predictions, one_hot_y)
        validation_accuracy = evaluate_accuracy(sess, X_valid_normalized, y_valid, predictions, one_hot_y)
        print("Cost at epoch {0} is {1}".format(epoch, current_cost))
        print("Train Accuracy = {:.3f}".format(train_accuracy))
        print("Validation Accuracy = {:.3f}".format(validation_accuracy))
    saver.save(sess, "/tmp/model.ckpt")
    writer.close()
    print('Model Saved')

Step 4 (Optional): Visualize the Neural Network's State with Test Images¶

This Section is not required to complete but acts as an additional excersise for understaning the output of a neural network's weights. While neural networks can be a great learning device they are often referred to as a black box. We can understand what the weights of a neural network look like better by plotting their feature maps. After successfully training your neural network you can see what it's feature maps look like by plotting the output of the network's weight layers in response to a test stimuli image. From these plotted feature maps, it's possible to see what characteristics of an image the network finds interesting. For a sign, maybe the inner network feature maps react with high activation to the sign's boundary outline or to the contrast in the sign's painted symbol.

Provided for you below is the function code that allows you to get the visualization output of any tensorflow weight layer you want. The inputs to the function should be a stimuli image, one used during training or a new one you provided, and then the tensorflow variable name that represents the layer's state during the training process, for instance if you wanted to see what the LeNet lab's feature maps looked like for it's second convolutional layer you could enter conv2 as the tf_activation variable.

For an example of what feature map outputs look like, check out NVIDIA's results in their paper End-to-End Deep Learning for Self-Driving Cars in the section Visualization of internal CNN State. NVIDIA was able to show that their network's inner weights had high activations to road boundary lines by comparing feature maps from an image with a clear path to one without. Try experimenting with a similar test to show that your trained network's weights are looking for interesting features, whether it's looking at differences in feature maps from images with or without a sign, or even what feature maps look like in a trained network vs a completely untrained one on the same sign image.
